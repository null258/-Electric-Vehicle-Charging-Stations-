import pandas as pd

df = pd.read_csv(
    r"C:\Users\本机\Downloads\archive\Electric_Vehicle_Charging_Stations.csv"
)

# 数据清洗
for col in [
    'EV Level1 EVSE Num',
    'EV Level2 EVSE Num',
    'EV DC Fast Count'
]:
    df[col] = pd.to_numeric(df[col], errors='coerce')

df.fillna(0, inplace=True)

# 新增总充电桩数
df['Total_EVSE'] = (
    df['EV Level1 EVSE Num']
    + df['EV Level2 EVSE Num']
    + df['EV DC Fast Count']
)

print(df.info())
print(df.describe())
